    def resolve_link(link,dir)
      case 
        when link =~ /^(http|ftp|mailto)/
          return nil
        when link =~ /^[\/a-zA-Z0-9_-]/
          return dir+'/'+link
        when link =~ /^\./
          return File.expand_path(dir+'/'+link)
        else
          return nil
      end
    end
