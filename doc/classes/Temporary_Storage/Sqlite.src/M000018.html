    def store_link(links)
      links.each do |link|
        if(links_in  = @db.get_first_value("select links_in from links where link = ?",link))
          @db.execute("update links set links_in =? where link = ? ",links_in+1,link)
        else
          @db.execute("insert into links values (?,?)", link, 1)
        end
      end
    end
