  def generate_files
    $logger.info("generating file base")
    out = Array.new
    out << "var files = {"
    @files.each_value do |f|
      out << "#{f.ID}:['#{f.title}','#{f.name}',#{f.page_rank}],"
    end
    @search_data_file.puts out.join[0..-2] + "}"    
  end
